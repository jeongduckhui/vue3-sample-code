function createFixedColumnDefs(rowData, headerName) {
  const firstRow = rowData?.[0];

  if (!firstRow) {
    return [];
  }

  const existingFields = FIXED_FIELDS.filter((field) =>
    Object.prototype.hasOwnProperty.call(firstRow, field)
  );

  return FIXED_COLUMN_DEFS.reduce((result, columnDef) => {
    // 1. 그룹 컬럼
    if (Array.isArray(columnDef.children)) {
      const existingChildren = columnDef.children
        .filter((child) => existingFields.includes(child.field))
        .map((child) => ({
          ...child,
          colId: child.colId ?? child.field,
          headerName: "",
        }));

      // 실제 데이터에 존재하는 자식 컬럼이 없으면 그룹도 제외
      if (existingChildren.length === 0) {
        return result;
      }

      result.push({
        ...columnDef,

        // App 또는 MODE를 그룹 헤더명으로 설정
        headerName:
          headerName === "APP" || headerName === "MODE"
            ? headerName
            : columnDef.headerName,

        children: existingChildren,
      });

      return result;
    }

    // 2. 일반 컬럼
    if (!existingFields.includes(columnDef.field)) {
      return result;
    }

    result.push({
      ...columnDef,
      colId: columnDef.colId ?? columnDef.field,
    });

    return result;
  }, []);
}
