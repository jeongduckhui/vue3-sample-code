const changedRows = {
  added: [],
  updated: [],
  deleted: []
};



const onCellValueChanged = (params) => {
  const row = params.data;

  if (!row._status) {
    row._status = "U"; // update
  }

  // 이미 있는지 체크
  const exists = changedRows.updated.find(r => r.id === row.id);
  if (!exists) {
    changedRows.updated.push(row);
  }
};
