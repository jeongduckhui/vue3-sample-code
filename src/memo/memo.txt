public interface MadCalendarRepositoryCustom {

    List<CalendarDto> findCalendar();
}




package com.example.demo.calendar.repository;

import com.example.demo.calendar.dto.CalendarDto;
import com.example.demo.calendar.entity.QMadCalendar;
import com.querydsl.core.types.dsl.Expressions;
import com.querydsl.jpa.impl.JPAQueryFactory;
import lombok.RequiredArgsConstructor;

import java.time.LocalDate;
import java.util.List;

@RequiredArgsConstructor
public class MadCalendarRepositoryImpl implements MadCalendarRepositoryCustom {

    private final JPAQueryFactory queryFactory;

    @Override
    public List<CalendarDto> findCalendar() {

        QMadCalendar calendar = QMadCalendar.madCalendar;

        // Java에서 계산 (실무 추천)
        String endYear = String.valueOf(
                LocalDate.now().plusMonths(60).getYear()
        );

        return queryFactory
                .select(
                        com.querydsl.core.types.Projections.constructor(
                                CalendarDto.class,
                                Expressions.stringTemplate(
                                        "substr({0}, 5, 6)",
                                        calendar.workYw
                                ),
                                calendar.wwSeq
                        )
                )
                .from(calendar)
                .where(
                        calendar.baseYear.between("2008", endYear)
                )
                .orderBy(
                        calendar.baseYear.asc(),
                        calendar.baseMm.asc(),
                        calendar.baseDay.asc()
                )
                .fetch();
    }
}

import com.example.demo.calendar.dto.CalendarDto;
import com.example.demo.calendar.repository.MadCalendarRepository;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

import java.util.List;

@Service
@RequiredArgsConstructor
public class CalendarService {

    private final MadCalendarRepository repository;

    public List<CalendarDto> getCalendar() {
        return repository.findCalendar();
    }
}
