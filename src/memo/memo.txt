import React, { useState } from "react";

const FindPopup = ({ onSearch, onClose }) => {
  const [keyword, setKeyword] = useState("");

  const [direction, setDirection] = useState("NEXT");
  const [position, setPosition] = useState("CURRENT");
  const [scope, setScope] = useState("ALL");
  const [condition, setCondition] = useState("CONTAINS");
  const [caseSensitive, setCaseSensitive] = useState(false);

  const handleSearch = () => {
    onSearch({
      keyword,
      direction,
      position,
      scope,
      condition,
      caseSensitive
    });
  };

  return (
    <div className="find-popup">
      <h3>찾기</h3>

      <input
        placeholder="검색어 입력"
        value={keyword}
        onChange={(e) => setKeyword(e.target.value)}
      />

      {/* 방향 */}
      <div>
        <b>찾을 방향</b>
        <label><input type="radio" checked={direction==="NEXT"} onChange={()=>setDirection("NEXT")} /> 다음</label>
        <label><input type="radio" checked={direction==="PREV"} onChange={()=>setDirection("PREV")} /> 이전</label>
      </div>

      {/* 위치 */}
      <div>
        <b>찾을 위치</b>
        <label><input type="radio" checked={position==="CURRENT"} onChange={()=>setPosition("CURRENT")} /> 현재위치</label>
        <label><input type="radio" checked={position==="FIRST"} onChange={()=>setPosition("FIRST")} /> 처음부터</label>
      </div>

      {/* 범위 */}
      <div>
        <b>찾을 범위</b>
        <label><input type="radio" checked={scope==="ALL"} onChange={()=>setScope("ALL")} /> 전체</label>
        <label><input type="radio" checked={scope==="ROW"} onChange={()=>setScope("ROW")} /> 행</label>
        <label><input type="radio" checked={scope==="COLUMN"} onChange={()=>setScope("COLUMN")} /> 열</label>
      </div>

      {/* 조건 */}
      <div>
        <b>찾을 조건</b>
        <label><input type="radio" checked={condition==="EQUAL"} onChange={()=>setCondition("EQUAL")} /> 일치</label>
        <label><input type="radio" checked={condition==="CONTAINS"} onChange={()=>setCondition("CONTAINS")} /> 포함</label>
      </div>

      {/* 대소문자 */}
      <div>
        <label>
          <input
            type="checkbox"
            checked={caseSensitive}
            onChange={() => setCaseSensitive(!caseSensitive)}
          />
          대/소문자 구분
        </label>
      </div>

      <button onClick={handleSearch}>찾기</button>
      <button onClick={onClose}>닫기</button>
    </div>
  );
};

export default FindPopup;



import { match } from "./findUtils";

export const useGridFind = (gridApiRef) => {

  const find = (option) => {
    const api = gridApiRef.current;
    if (!api) return;

    const focusedCell = api.getFocusedCell();
    if (!focusedCell) return;

    const rowCount = api.getDisplayedRowCount();
    const colDefs = api.getColumnDefs();

    let rowIndex =
      option.position === "FIRST" ? 0 : focusedCell.rowIndex;

    const step = option.direction === "NEXT" ? 1 : -1;

    while (rowIndex >= 0 && rowIndex < rowCount) {
      const rowNode = api.getDisplayedRowAtIndex(rowIndex);

      let columns = colDefs;

      if (option.scope === "COLUMN") {
        columns = [focusedCell.column];
      }

      if (option.scope === "ROW") {
        columns = colDefs;
      }

      for (let col of columns) {
        const value = rowNode.data[col.field];

        if (match(value, option)) {
          api.ensureIndexVisible(rowIndex);
          api.setFocusedCell(rowIndex, col.field);
          return;
        }
      }

      rowIndex += step;
    }

    alert("찾을 수 없습니다.");
  };

  return { find };
};




export const match = (value, option) => {
  if (value == null) return false;

  let target = String(value);
  let keyword = option.keyword;

  if (!option.caseSensitive) {
    target = target.toLowerCase();
    keyword = keyword.toLowerCase();
  }

  if (option.condition === "EQUAL") {
    return target === keyword;
  }

  if (option.condition === "CONTAINS") {
    return target.includes(keyword);
  }

  return false;
};



export const getContextMenuItems = (params, openFindPopup) => {
  return [
    {
      name: "찾기",
      action: () => openFindPopup(params)
    }
  ];
};


import React, { useRef, useState } from "react";
import { AgGridReact } from "ag-grid-react";
import FindPopup from "./find/FindPopup";
import { useGridFind } from "./find/useGridFind";
import { getContextMenuItems } from "./find/gridContextMenu";

const GridPage = () => {
  const gridApiRef = useRef(null);
  const [showPopup, setShowPopup] = useState(false);

  const { find } = useGridFind(gridApiRef);

  const openFindPopup = () => {
    setShowPopup(true);
  };

  return (
    <div style={{ height: 500 }}>
      <AgGridReact
        onGridReady={(params) => {
          gridApiRef.current = params.api;
        }}
        getContextMenuItems={(params) =>
          getContextMenuItems(params, openFindPopup)
        }
      />

      {showPopup && (
        <FindPopup
          onSearch={(option) => find(option)}
          onClose={() => setShowPopup(false)}
        />
      )}
    </div>
  );
};

export default GridPage;
