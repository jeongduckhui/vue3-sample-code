import { AgGridReact } from "ag-grid-react";
import { useState, useRef } from "react";

import "ag-grid-community/styles/ag-grid.css";
import "ag-grid-community/styles/ag-theme-alpine.css";

function GridExample() {

  const gridRef = useRef();

  const [rowData, setRowData] = useState([
    {
      all:false,
      colA:false,
      colB:false,
      colC:false,

      all2:false,
      colD:false,
      colE:false,
      colF:false
    }
  ]);

  const columnDefs = [

    { field:"all", headerName:"ALL",
      cellRenderer:"agCheckboxCellRenderer",
      editable:true
    },

    { field:"colA", headerName:"A",
      cellRenderer:"agCheckboxCellRenderer",
      editable:true
    },

    { field:"colB", headerName:"B",
      cellRenderer:"agCheckboxCellRenderer",
      editable:true
    },

    { field:"colC", headerName:"C",
      cellRenderer:"agCheckboxCellRenderer",
      editable:true
    },


    { field:"all2", headerName:"ALL2",
      cellRenderer:"agCheckboxCellRenderer",
      editable:true
    },

    { field:"colD", headerName:"D",
      cellRenderer:"agCheckboxCellRenderer",
      editable:true
    },

    { field:"colE", headerName:"E",
      cellRenderer:"agCheckboxCellRenderer",
      editable:true
    },

    { field:"colF", headerName:"F",
      cellRenderer:"agCheckboxCellRenderer",
      editable:true
    }

  ];


  const onCellValueChanged = (params)=>{

    const field = params.colDef.field;
    const checked = params.newValue;
    const node = params.node;

    // ALL -> 2,3,4
    if(field==="all"){
      node.setDataValue("colA", checked);
      node.setDataValue("colB", checked);
      node.setDataValue("colC", checked);
    }

    // ALL2 -> 6,7,8
    if(field==="all2"){
      node.setDataValue("colD", checked);
      node.setDataValue("colE", checked);
      node.setDataValue("colF", checked);
    }

  };


  return (

    <div className="ag-theme-alpine" style={{height:400}}>

      <AgGridReact
        ref={gridRef}
        rowData={rowData}
        columnDefs={columnDefs}
        onCellValueChanged={onCellValueChanged}
      />

    </div>

  );

}

export default GridExample;
