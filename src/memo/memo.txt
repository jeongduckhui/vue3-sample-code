import java.io.Serializable;
import java.util.Objects;

public class SampleId implements Serializable {

    private String userId;
    private String roleId;

    // 기본 생성자 필수
    public SampleId() {}

    public SampleId(String userId, String roleId) {
        this.userId = userId;
        this.roleId = roleId;
    }

    // equals & hashCode 필수
    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;

        SampleId that = (SampleId) o;
        return Objects.equals(userId, that.userId)
            && Objects.equals(roleId, that.roleId);
    }

    @Override
    public int hashCode() {
        return Objects.hash(userId, roleId);
    }
}

package com.example.demo.user.repository;

import com.example.demo.user.dto.UserDto;
import com.example.demo.user.dto.UserSearchCond;
import com.example.demo.user.entity.QDept;
import com.example.demo.user.entity.QUser;
import com.querydsl.core.types.Projections;
import com.querydsl.core.types.dsl.*;
import com.querydsl.jpa.impl.JPAQueryFactory;
import lombok.RequiredArgsConstructor;

import java.util.List;

@RequiredArgsConstructor
public class UserRepositoryImpl implements UserRepositoryCustom {

    private final JPAQueryFactory queryFactory;

    @Override
    public List<UserDto> search(UserSearchCond cond) {

        QUser user = QUser.user;
        QDept dept = QDept.dept;

        return queryFactory
                .select(Projections.constructor(
                        UserDto.class,
                        user.userId,
                        user.userName,
                        dept.deptName,

                        // NVL
                        user.age.coalesce(0),

                        // DECODE
                        new CaseBuilder()
                                .when(user.status.eq("A")).then("ACTIVE")
                                .when(user.status.eq("I")).then("INACTIVE")
                                .otherwise("UNKNOWN")
                ))
                .from(user)
                .join(dept).on(user.deptId.eq(dept.deptId))
                .where(
                        nameLike(cond.name, user),
                        deptIn(cond.deptIds, user),
                        statusEq(cond.status, user)
                )
                .orderBy(user.createdAt.desc())
                .fetch();
    }

    private BooleanExpression nameLike(String name, QUser user) {
        return (name == null || name.isEmpty()) ? null : user.userName.contains(name);
    }

    private BooleanExpression deptIn(List<String> deptIds, QUser user) {
        return (deptIds == null || deptIds.isEmpty()) ? null : user.deptId.in(deptIds);
    }

    private BooleanExpression statusEq(String status, QUser user) {
        return (status == null || status.isEmpty()) ? null : user.status.eq(status);
    }
}
