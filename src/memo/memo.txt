function buildExpandColumnsFromMMWW(columns) {
  const result = [];
  let currentGroupKey = null;

  columns.forEach(({ code }) => {
    // 부모 MM
    if (code.startsWith('MM')) {
      currentGroupKey = code;
      return;
    }

    // 자식 WW
    if (code.startsWith('WW') && currentGroupKey) {
      result.push({
        groupKey: currentGroupKey,
        children: code,
      });
    }
  });

  return result;
}
