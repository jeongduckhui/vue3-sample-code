public class RequestUtils {

    public static HttpServletRequest getRequest() {
        ServletRequestAttributes attrs =
            (ServletRequestAttributes) RequestContextHolder.getRequestAttributes();

        return (attrs != null) ? attrs.getRequest() : null;
    }

    public static String getHeader(String name) {
        HttpServletRequest request = getRequest();
        return (request != null) ? request.getHeader(name) : null;
    }
}






package com.example.demo.cache.support;

import com.github.benmanes.caffeine.cache.Cache;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

import java.util.Optional;
import java.util.function.Supplier;

@Slf4j
@Component
public class CaffeineCacheSupport {

    /**
     * 캐시 조회
     */
    public <K, V> Optional<V> get(Cache<K, V> cache, K key) {
        V value = cache.getIfPresent(key);

        if (value != null) {
            log.debug("CACHE HIT key={}", key);
        } else {
            log.debug("CACHE MISS key={}", key);
        }

        return Optional.ofNullable(value);
    }

    /**
     * 캐시 조회 + 없으면 로딩
     */
    public <K, V> V getOrLoad(Cache<K, V> cache, K key, Supplier<V> loader) {

        V value = cache.getIfPresent(key);

        if (value != null) {
            log.debug("CACHE HIT key={}", key);
            return value;
        }

        log.debug("CACHE MISS → DB LOAD key={}", key);

        V loadedValue = loader.get();

        if (loadedValue != null) {
            cache.put(key, loadedValue);
        }

        return loadedValue;
    }

    /**
     * 캐시 저장
     */
    public <K, V> void put(Cache<K, V> cache, K key, V value) {
        log.debug("CACHE PUT key={}", key);
        cache.put(key, value);
    }

    /**
     * 캐시 삭제
     */
    public <K, V> void evict(Cache<K, V> cache, K key) {
        log.debug("CACHE EVICT key={}", key);
        cache.invalidate(key);
    }

    /**
     * 전체 삭제
     */
    public <K, V> void evictAll(Cache<K, V> cache) {
        log.debug("CACHE EVICT ALL");
        cache.invalidateAll();
    }
}






package com.example.demo.cache.support;

import com.github.benmanes.caffeine.cache.Cache;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

import java.util.Optional;
import java.util.function.Supplier;

@Component
@RequiredArgsConstructor
public class CaffeineCacheSupport {

    /**
     * 캐시 조회
     */
    public <K, V> Optional<V> get(Cache<K, V> cache, K key) {
        return Optional.ofNullable(cache.getIfPresent(key));
    }

    /**
     * 캐시 조회 + 없으면 로딩
     */
    public <K, V> V getOrLoad(Cache<K, V> cache, K key, Supplier<V> loader) {
        return cache.get(key, k -> loader.get());
    }

    /**
     * 캐시 저장
     */
    public <K, V> void put(Cache<K, V> cache, K key, V value) {
        cache.put(key, value);
    }

    /**
     * 캐시 삭제
     */
    public <K, V> void evict(Cache<K, V> cache, K key) {
        cache.invalidate(key);
    }

    /**
     * 전체 삭제
     */
    public <K, V> void evictAll(Cache<K, V> cache) {
        cache.invalidateAll();
    }
}


@Service
@RequiredArgsConstructor
public class CommonCodeCacheService {

    private final Cache<String, Object> cache;
    private final CaffeineCacheSupport cacheSupport;

    public Object get(String key) {
        return cacheSupport.getOrLoad(cache, key, () -> loadFromDb(key));
    }
}
