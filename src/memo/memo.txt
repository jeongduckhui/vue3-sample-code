const columnDefs = [
  {
    headerName: '내용',
    field: 'content',
    cellRenderer: RowResizeCellRenderer,
  },
];



import React, { useRef } from 'react';

const RowResizeCellRenderer = (props) => {
  const startYRef = useRef(0);
  const startHeightRef = useRef(0);

  const onMouseDown = (e) => {
    e.preventDefault();

    startYRef.current = e.clientY;
    startHeightRef.current = props.node.rowHeight;

    document.addEventListener('mousemove', onMouseMove);
    document.addEventListener('mouseup', onMouseUp);
  };

  const onMouseMove = (e) => {
    const diff = e.clientY - startYRef.current;
    const newHeight = Math.max(24, startHeightRef.current + diff);

    props.api.setRowHeight(props.node, newHeight);
    props.api.onRowHeightChanged();
  };

  const onMouseUp = () => {
    document.removeEventListener('mousemove', onMouseMove);
    document.removeEventListener('mouseup', onMouseUp);
  };

  return (
    <div style={{ position: 'relative', height: '100%' }}>
      {props.value}

      <div
        onMouseDown={onMouseDown}
        style={{
          position: 'absolute',
          bottom: 0,
          left: 0,
          right: 0,
          height: 6,
          cursor: 'row-resize',
          background: 'transparent',
        }}
      />
    </div>
  );
};

export default RowResizeCellRenderer;
